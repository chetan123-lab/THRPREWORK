provider "aws" {
  region = var.aws_region
}

# Check if the key pair exists
data "aws_key_pair" "example" {
  key_name = var.key_name
}

# Create the key pair if it doesn't exist
resource "tls_private_key" "example" {
  algorithm = "RSA"
  rsa_bits  = 4096
}

resource "aws_key_pair" "example" {
  count     = try(data.aws_key_pair.example.id, "") == "" ? 1 : 0
  key_name   = var.key_name
  public_key = tls_private_key.example.public_key_openssh
}

# Check if the security group exists
data "aws_security_group" "example" {
  filter {
    name   = "group-name"
    values = [var.security_group_name]
  }
}

# Create the security group if it doesn't exist
resource "aws_security_group" "example" {
  count       = length(data.aws_security_group.example.id) > 0 ? 0 : 1
  name        = var.security_group_name
  description = "My security group"

  ingress {
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

# Create an EC2 instance
resource "aws_instance" "example" {
  ami           = var.ami_id
  instance_type = var.instance_type
  key_name      = var.key_name
  vpc_security_group_ids = length(aws_security_group.example) > 0 ? [aws_security_group.example[0].id] : [data.aws_security_group.example.id]

  tags = {
    Name = "example-instance"
  }
}
